1. SELEZIONARE TUTTI GLI STUDENTI NATI NEL 1990

FROM students

3. SELEZIONARE TUTTI GLI STUDENTI CHE HANNO PIU DI 30 ANNI

SELECT *
FROM students

4. SELEZIONARE TUTTI I CORSI DEL PRIMO SEMESSTRE DEL PRIMO ANNO DI UN QUALSIASI CORSO DI LAUREA

SELECT *
FROM courses
WHERE period = 'I semestre' AND year LIKE 1;
